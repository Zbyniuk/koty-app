import React, { useEffect, useMemo, useState } from "react";
import { View, Text, Image, StyleSheet, TouchableOpacity, ScrollView } from "react-native";
import { useLocalSearchParams, router } from "expo-router";
import { supabase } from "../../lib/supabase";

function monthsBetween(dateString?: string | null) {
  if (!dateString) return null;
  const birth = new Date(dateString);
  if (Number.isNaN(birth.getTime())) return null;

  const now = new Date();
  let months = (now.getFullYear() - birth.getFullYear()) * 12;
  months += now.getMonth() - birth.getMonth();

  if (now.getDate() < birth.getDate()) {
    months -= 1;
  }

  return months;
}

function shouldShowCallName(name?: string | null, callName?: string | null) {
  const n = (name || "").trim();
  const c = (callName || "").trim();

  if (!c) return false;
  if (!n) return true;

  const firstWord = n.split(/\s+/)[0]?.trim().toLowerCase() || "";
  const normalizedCall = c.toLowerCase();

  return normalizedCall !== firstWord;
}

export default function CatScreen() {
  const { id } = useLocalSearchParams();

  const [cat, setCat] = useState<any>(null);
  const [motherLitterCount, setMotherLitterCount] = useState(0);
  const [fatherLitterCount, setFatherLitterCount] = useState(0);

  useEffect(() => {
    loadCat();
  }, []);

  const loadCat = async () => {
    const { data } = await supabase.from("cats").select("*").eq("id", id).single();
    setCat(data || null);

    const { count: motherCount } = await supabase
      .from("litters")
      .select("*", { count: "exact", head: true })
      .eq("mother_id", id);

    const { count: fatherCount } = await supabase
      .from("litters")
      .select("*", { count: "exact", head: true })
      .eq("father_id", id);

    setMotherLitterCount(motherCount || 0);
    setFatherLitterCount(fatherCount || 0);
  };

  const displayName = useMemo(() => {
    if (!cat) return "Bez nazwy";
    return cat.name || "Bez nazwy";
  }, [cat]);

  const callNameLine = useMemo(() => {
    if (!cat?.call_name) return "";
    return `(${cat.call_name})`;
  }, [cat]);

  const titleLine = useMemo(() => {
    if (!cat) return "";
    return [cat.title_prefix || "", cat.title_suffix || ""]
      .map((x: string) => (x || "").trim())
      .filter(Boolean)
      .join(" ");
  }, [cat]);

  const showBreedingTile = useMemo(() => {
    if (!cat) return false;

    const ageMonths = monthsBetween(cat.birth_date);
    if (ageMonths !== null && ageMonths < 11) return false;
    return true;
  }, [cat]);

  const breedingTileConfig = useMemo(() => {
    if (!cat || !showBreedingTile) return null;

    const isFemale = cat.sex === "F";
    const isMale = cat.sex === "M";
    const status = (cat.status || "").toLowerCase();

    if (isFemale) {
      const shouldShow = status === "breeding" || motherLitterCount > 0;
      if (!shouldShow) return null;

      return {
        label: "Mioty",
        route: `/litter-list/${id}`,
        color: "#b56576",
      };
    }

    if (isMale) {
      const shouldShow = status === "breeding" || fatherLitterCount > 0;
      if (!shouldShow) return null;

      return {
        label: "Krycia",
        route: `/litter-list/${id}`,
        color: "#b56576",
      };
    }

    return null;
  }, [cat, showBreedingTile, motherLitterCount, fatherLitterCount, id]);

  if (!cat) {
    return (
      <View style={styles.loadingWrap}>
        <Text>Ładowanie...</Text>
      </View>
    );
  }

  return (
    <ScrollView style={styles.container}>
      <View style={styles.topRow}>
        {cat.photo_url ? (
          <Image source={{ uri: cat.photo_url }} style={styles.photo} />
        ) : (
          <View style={[styles.photo, styles.photoPlaceholder]}>
            <Text style={styles.photoPlaceholderText}>Brak zdjęcia</Text>
          </View>
        )}

        <View style={styles.right}>
          {titleLine ? <Text style={styles.title}>{titleLine}</Text> : null}

          <Text
            style={styles.nameSingleLine}
            numberOfLines={2}
            ellipsizeMode="tail"
          >
            {displayName}
          </Text>

          {callNameLine ? (
            <Text style={styles.callName}>{callNameLine}</Text>
          ) : null}

          {cat.birth_date ? (
            <Text style={styles.birthDate}>ur. {cat.birth_date}</Text>
          ) : null}
        </View>
      </View>

      <View style={styles.grid}>
        <TouchableOpacity
          style={[styles.tile, { backgroundColor: "#6d597a" }]}
          onPress={() => router.push(`/cat-details/${id}`)}
        >
          <Text style={styles.tileText}>Szczegóły</Text>
        </TouchableOpacity>

        {breedingTileConfig ? (
          <TouchableOpacity
            style={[styles.tile, { backgroundColor: breedingTileConfig.color }]}
            onPress={() => router.push(breedingTileConfig.route)}
          >
            <Text style={styles.tileText}>{breedingTileConfig.label}</Text>
          </TouchableOpacity>
        ) : null}

        <TouchableOpacity
          style={[styles.tile, { backgroundColor: "#355070" }]}
          onPress={() => router.push(`/cat-documents/${id}`)}
        >
          <Text style={styles.tileText}>Dokumenty</Text>
        </TouchableOpacity>

        <TouchableOpacity
          style={[styles.tile, { backgroundColor: "#6a994e" }]}
          onPress={() => router.push(`/cat-health/${id}`)}
        >
          <Text style={styles.tileText}>Zdrowie</Text>
        </TouchableOpacity>
      </View>
    </ScrollView>
  );
}
